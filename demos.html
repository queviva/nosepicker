so you think you need a more complicated colorpicker? fine then ...

nosepicker.js is totally complicated, and bloated, and ridiculous, and allows for all kinds of customizations - it's the perfect thing for you picky nosepickers
include the script, along with any options in the data-param; specify nosepicker elements with a data-param; handle any events with listeners, as usual; set any options dynamically by dispatching an event to the nosepicker: 

<script src="nosepicker.js"
 data-nose='{
    "selector"  : 'nose',
    "kind"      : 'touch,wheel', // wheel|touch|swipe|coast|point
    "colorsrc"  : 'background-color',
    "loaded"    : 'nose-loaded',
    "input"     : 'nose-input',
    "colorSelf" : true,
    "colorText" : true,
    "pattern"   : true,
    "able"      : true,
    "makeRef"   : false,
    "sens"      : { X: 1, Y: 1 }
 '}
></script>

<div id="picker" data-nose></div>

<script>
    
    picker.addEventListener('nose-input', e => {
    
        // e.detail.value is an hsla color string
        
        // e.detail.hsla is an object ...
        // e.detail.hsla.h is the hue
        // e.detail.hsla.s is the saturation
        // e.detail.hsla.l is the luminosity
        // e.detail.hsla.a is the opacity
        
        // e.detail.vals is an array ...
        // e.detail.vals[0] is the hue
        // e.detail.vals[1] is the saturation
        // e.detail.vals[2] is the luminosity
        // e.detail.vals[3] is the opacity
        
    });
    
    // set options dynamically by dispatching events ...
    // settable options are value|kind|able|sens
    
    picker.dispatchEvent(new CustomEvent(
        'nose-set-value', {detail:'0,100,50,1'})
    );
    
    picker.dispatchEvent(new CustomEvent(
        'nose-set-kind', {detail: 'coast,point'})
    );
    
    picker.dispatchEvent(new CustomEvent(
        'nose-set-able', {detail: false})
    );
    
    picker.dispatchEvent(new CustomEvent(
        'nose-set-sens', {detail: {X:1, Y:1})
    );
    
</script>
